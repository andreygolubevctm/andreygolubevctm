USE ctm;

-- Get Provider ID from Provider Code
SET @provider_id = (SELECT providerId FROM ctm.provider_master WHERE providerCode = '1');

-- Create temporary table of product ids
CREATE TEMPORARY TABLE _products SELECT productid FROM ctm.product_master WHERE productCat='CREDITCARD' AND providerid=@provider_id;

-- Should return less than 12 products for most providers
SELECT * FROM _products;

-- Delete using join with temporary table
DELETE cpm FROM ctm.category_product_mapping cpm INNER JOIN _products ON cpm.productID=_products.productId;
DELETE pp FROM ctm.product_properties pp INNER JOIN _products ON pp.productID=_products.productId;
DELETE ppt FROM ctm.product_properties_text ppt INNER JOIN _products ON ppt.productID=_products.productId;
DELETE pm FROM ctm.product_master pm INNER JOIN _products ON pm.productID=_products.productId;

-- Clean up
DROP TEMPORARY TABLE _products;
